Configure o Google Apps Script para acessar os 200.000+ ativos

A Base de Ativos é grande demais para embutir no HTML. O dashboard busca em tempo real via Google Apps Script, retornando só os ativos que batem com seus filtros. Setup leva 5 minutos e é único.

⚠️ Falha ao buscar ativos do Apps Script

A chamada falhou. Causas comuns:

  • URL errada: deve terminar em /exec (não /edit)
  • Implantação privada: acesso deve ser "Qualquer pessoa" — refaça a implantação se necessário
  • Sheet ID errado: no Apps Script, verifique a constante SHEET_ID
  • Aba não encontrada: verifique se a aba se chama exatamente Base de Ativos
  • Permissões não autorizadas: rode o script manualmente uma vez no editor para autorizar

Configurar Google Apps Script (API)

Setup único de 5 minutos. Ganha duas funcionalidades: (1) criar planilhas do Sheets com os dados filtrados, e (2) acessar os 200.000+ ativos individuais em tempo real na aba "Lista Analítica".

1
Abrir o Google Apps Script

Acesse script.google.com/create e faça login.

2
Colar o código abaixo

Apague o conteúdo existente e cole:

// ═══════════════════════════════════════════════════════════ // API DASHBOARD INVESTIMENTOS MLB // POST: cria planilha do dashboard | GET: busca ativos // ═══════════════════════════════════════════════════════════ var SHEET_ID = '1uD2YMCB0p4hRbq-9JP9U6qMha0Kb8APXBQRjt2GuRZY'; var ASSETS_SHEET = 'Base de Ativos'; function doPost(e) { var csv = e.parameter.data || e.postData.contents; var rows = Utilities.parseCsv(csv); var date = Utilities.formatDate(new Date(), 'America/Sao_Paulo', 'dd/MM/yyyy HH:mm'); var ss = SpreadsheetApp.create('Investimentos MLB - ' + date); var sheet = ss.getActiveSheet(); sheet.setName('Dados'); if (rows.length > 0) { sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows); var hdr = sheet.getRange(1, 1, 1, rows[0].length); hdr.setBackground('#1a1a2e').setFontColor('#FFF200').setFontWeight('bold'); sheet.setFrozenRows(1); if (rows[0].length >= 11) sheet.getRange(2, 7, rows.length-1, 5).setNumberFormat('#,##0.00'); sheet.autoResizeColumns(1, rows[0].length); } var url = ss.getUrl(); return HtmlService.createHtmlOutput('<script>window.location.href="'+url+'";</' + 'script><p>Abrindo planilha... <a href="'+url+'">Clique aqui</a></p>'); } function doGet(e) { var callback = e.parameter.callback; var action = e.parameter.action || 'list'; var result; try { if (action === 'list') result = listAssets(e.parameter); else result = {error: 'Unknown action: ' + action}; } catch(err) { result = {error: err.toString()}; } if (callback) { return ContentService.createTextOutput(callback + '(' + JSON.stringify(result) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function listAssets(p) { var sh = SpreadsheetApp.openById(SHEET_ID).getSheetByName(ASSETS_SHEET); var data = sh.getDataRange().getValues(); var headers = data.shift(); var idx = {}; headers.forEach(function(h, i) { idx[String(h).trim()] = i; }); var fc=idx['FC'], pt=idx['PT'], est=idx['Estado']; var af=idx['AF'], desc=idx['DENOMINACION_AF']; var compra=idx['MONTO_COMPRA_LC'], depr=idx['MONTO_DEPRECIACION_LC']; var contabil=idx['MONTO_CONTABLE_LC']; var fecha=idx['FECHA_ALTA'], vu=idx['VU_MESES_AF']; var fcF = p.fc ? p.fc.split('|').filter(Boolean) : null; var ptF = p.pt ? p.pt.split('|').filter(Boolean) : null; var estF = p.estado ? p.estado.split('|').filter(Boolean) : null; var q = (p.q || '').toLowerCase(); var sortBy = p.sort || 'compra_desc'; var limit = Math.min(parseInt(p.limit || '500'), 5000); var offset = parseInt(p.offset || '0'); var filtered = [], totalCompra = 0, totalContabil = 0; for (var i = 0; i < data.length; i++) { var r = data[i]; if (fcF && fcF.indexOf(String(r[fc])) < 0) continue; if (ptF && ptF.indexOf(String(r[pt])) < 0) continue; if (estF && estF.indexOf(String(r[est])) < 0) continue; if (q) { var hay = ((r[af]||'') + ' ' + (r[desc]||'') + ' ' + (r[fc]||'') + ' ' + (r[est]||'')).toLowerCase(); if (hay.indexOf(q) < 0) continue; } var c = parseFloat(r[compra]) || 0; var ct = parseFloat(r[contabil]) || 0; totalCompra += c; totalContabil += ct; filtered.push({ af: r[af], desc: r[desc], compra: c, depr: parseFloat(r[depr]) || 0, contabil: ct, fecha: r[fecha] instanceof Date ? Utilities.formatDate(r[fecha], 'America/Sao_Paulo', 'dd/MM/yyyy') : String(r[fecha]||''), pt: r[pt], fc: r[fc], vu: String(r[vu]||''), estado: r[est] }); } filtered.sort(function(a, b) { if (sortBy === 'compra_desc') return b.compra - a.compra; if (sortBy === 'compra_asc') return a.compra - b.compra; if (sortBy === 'contabil_desc') return b.contabil - a.contabil; if (sortBy === 'fc_asc') return String(a.fc).localeCompare(String(b.fc)); return b.compra - a.compra; }); var top = filtered.length > 0 ? filtered[0] : null; var paged = filtered.slice(offset, offset + limit); return { total: filtered.length, totalCompra: totalCompra, totalContabil: totalContabil, offset: offset, limit: limit, top: top, rows: paged }; }
